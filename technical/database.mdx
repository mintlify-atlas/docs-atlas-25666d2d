## Database Schema

```plaintext
firestore/
├── users/
│   └── {userId}
│       ├── uid: string
│       ├── email: string
│       ├── displayName: string
│       ├── photoURL: string
│       ├── createdAt: timestamp
│       ├── role: string
│       └── preferences: object
│
├── pets/
│   └── {petId}
│       ├── userId: string
│       ├── type: string (lost|found|adoption)
│       ├── petType: string (dog|cat|other)
│       ├── name: string
│       ├── description: string
│       ├── photos: array
│       ├── location: geopoint
│       ├── address: string
│       ├── contactPhone: string
│       ├── contactWhatsApp: string
│       ├── status: string (active|resolved|closed)
│       ├── createdAt: timestamp
│       └── updatedAt: timestamp
│
├── messages/
│   └── {messageId}
│       ├── petId: string
│       ├── senderId: string
│       ├── receiverId: string
│       ├── message: string
│       ├── read: boolean
│       └── createdAt: timestamp
│
└── reports/
    └── {reportId}
        ├── petId: string
        ├── reportedBy: string
        ├── reason: string
        ├── description: string
        ├── status: string (pending|reviewed|resolved)
        └── createdAt: timestamp
```

## Collection Schemas

<Accordion title="Users Collection">
  ### Schema Definition
  
  ```typescript
  interface User {
    uid: string                    // Firebase Auth UID
    email: string                  // User email address
    displayName: string            // User display name
    photoURL: string | null        // Profile photo URL
    createdAt: Timestamp           // Account creation date
    emailVerified: boolean         // Email verification status
    role: 'user' | 'admin'        // User role
    authProvider: 'email' | 'google' | 'facebook'
    preferences: {
      notifications: boolean       // Push notification preference
      emailUpdates: boolean        // Email updates preference
      language: string             // User language (es by default)
    }
    stats?: {
      reportsCreated: number       // Total reports created
      petsReunited: number         // Successful reunions
    }
  }
  ```

  ### Example Document
  
  ```javascript
  {
    uid: "abc123xyz",
    email: "maria@example.com",
    displayName: "María González",
    photoURL: "https://storage.googleapis.com/...",
    createdAt: Timestamp.now(),
    emailVerified: true,
    role: "user",
    authProvider: "google",
    preferences: {
      notifications: true,
      emailUpdates: true,
      language: "es"
    },
    stats: {
      reportsCreated: 3,
      petsReunited: 1
    }
  }
  ```

  ### Indexes
  - `email` (ascending)
  - `createdAt` (descending)
  - `role` (ascending)

## Real-time Listeners

<CodeGroup>
  ```javascript useRealtimePets.js
  import { useEffect, useState } from 'react'
  import { collection, query, where, orderBy, onSnapshot } from 'firebase/firestore'
  import { db } from '../config/firebase'

  export const useRealtimePets = (type, status = 'active') => {
    const [pets, setPets] = useState([])
    const [loading, setLoading] = useState(true)
    const [error, setError] = useState(null)

    useEffect(() => {
      const q = query(
        collection(db, 'pets'),
        where('type', '==', type),
        where('status', '==', status),
        orderBy('createdAt', 'desc')
      )

      const unsubscribe = onSnapshot(
        q,
        (snapshot) => {
          const petsData = []
          snapshot.forEach((doc) => {
            petsData.push({ id: doc.id, ...doc.data() })
          })
          setPets(petsData)
          setLoading(false)
        },
        (err) => {
          console.error('Error in realtime listener:', err)
          setError(err)
          setLoading(false)
        }
      )

      return () => unsubscribe()
    }, [type, status])

    return { pets, loading, error }
  }
  ```

  ```javascript usePetMessages.js
  import { useEffect, useState } from 'react'

        })
        setMessages(msgs)
        setLoading(false)
      })

      return () => unsubscribe()
    }, [petId])

    return { messages, loading }
  }
  ```

## Geolocation Queries

<Note>
  Firestore doesn't support radius queries natively. Use the geofire-common library for geospatial queries.
</Note>

```javascript
import { collection, query, where, orderBy, startAt, endAt, getDocs } from 'firebase/firestore'
import { geohashQueryBounds, distanceBetween } from 'geofire-common'
import { db } from '../config/firebase'

// Find pets within radius (in kilometers)
export const findPetsNearby = async (lat, lng, radiusInKm) => {
  const center = [lat, lng]
  const bounds = geohashQueryBounds(center, radiusInKm * 1000) // Convert to meters
  
  const promises = []
  
  for (const b of bounds) {
    const q = query(
      collection(db, 'pets'),
      orderBy('geohash'),
      startAt(b[0]),
      endAt(b[1])
    )
    
    promises.push(getDocs(q))
  }
  
  const snapshots = await Promise.all(promises)
  const matchingDocs = []
  
  for (const snap of snapshots) {
    for (const doc of snap.docs) {
      const data = doc.data()
      const lat = data.location.latitude
      const lng = data.location.longitude
      
      // Filter by actual distance
      const distanceInKm = distanceBetween([lat, lng], center)
      
      if (distanceInKm <= radiusInKm) {
        matchingDocs.push({
          id: doc.id,
          ...data,
          distance: distanceInKm
        })
      }
    }
  }
  
  // Sort by distance
  matchingDocs.sort((a, b) => a.distance - b.distance)
  
  return matchingDocs
}
```

## Firestore Security Rules

```javascript
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function isOwner(userId) {
      return request.auth.uid == userId;
    }
    
    function isAdmin() {
      return isAuthenticated() && 
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    
    // Users collection
    match /users/{userId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && isOwner(userId);
      allow update: if isOwner(userId) || isAdmin();
      allow delete: if isAdmin();
    }
    
    // Pets collection
    match /pets/{petId} {
      allow read: if true; // Public read access
      allow create: if isAuthenticated() && 
                      request.resource.data.userId == request.auth.uid;
      allow update: if isAuthenticated() && 
                      (resource.data.userId == request.auth.uid || isAdmin());
      allow delete: if isAuthenticated() && 
                      (resource.data.userId == request.auth.uid || isAdmin());
    }
    
    // Messages collection
    match /messages/{messageId} {
      allow read: if isAuthenticated() && 
                    (resource.data.senderId == request.auth.uid || 
                     resource.data.receiverId == request.auth.uid);
      allow create: if isAuthenticated() && 
                      request.resource.data.senderId == request.auth.uid;
      allow update: if isAuthenticated() && 
                      resource.data.receiverId == request.auth.uid; // Only for marking as read
      allow delete: if isAdmin();
    }
    
    // Reports collection
    match /reports/{reportId} {
      allow read: if isAdmin();
      allow create: if isAuthenticated();
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }
  }
}
```

  Always test security rules thoroughly before deploying to production. Use the Firebase Console Rules Playground to simulate requests.
</Warning>

## Pagination

<CodeGroup>
  ```javascript usePagination.js
  import { useState } from 'react'
  import { collection, query, orderBy, limit, startAfter, getDocs } from 'firebase/firestore'
  import { db } from '../config/firebase'

  export const usePagination = (collectionName, pageSize = 10) => {
    const [documents, setDocuments] = useState([])
    const [lastDoc, setLastDoc] = useState(null)
    const [loading, setLoading] = useState(false)
    const [hasMore, setHasMore] = useState(true)

    const loadMore = async () => {
      if (loading || !hasMore) return

      setLoading(true)

      try {
        let q = query(
          collection(db, collectionName),
          orderBy('createdAt', 'desc'),
          limit(pageSize)
        )

        if (lastDoc) {
          q = query(q, startAfter(lastDoc))
        }

        const snapshot = await getDocs(q)
        
        if (snapshot.empty) {
          setHasMore(false)
          setLoading(false)
          return
        }

        const newDocs = []
        snapshot.forEach((doc) => {
          newDocs.push({ id: doc.id, ...doc.data() })
        })

        setDocuments((prev) => [...prev, ...newDocs])
        setLastDoc(snapshot.docs[snapshot.docs.length - 1])
        setHasMore(snapshot.docs.length === pageSize)
      } catch (error) {
        console.error('Pagination error:', error)
      }

      setLoading(false)
    }

    const reset = () => {
      setDocuments([])
      setLastDoc(null)
      setHasMore(true)
    }

    return { documents, loading, hasMore, loadMore, reset }
  }
  ```
</CodeGroup>

## Data Migration

<Tip>
  Use Firebase Admin SDK for batch operations and data migrations to avoid client-side rate limits.
</Tip>

```javascript
// Node.js script for data migration
const admin = require('firebase-admin')
const serviceAccount = require('./serviceAccountKey.json')

admin.initializeApp({
  credential: admin.credential.cert(serviceAccount)
})

const db = admin.firestore()

// Batch update example
async function addGeohashToPets() {
  const { geohashForLocation } = require('geofire-common')
  const petsRef = db.collection('pets')
  const snapshot = await petsRef.get()
  
  const batch = db.batch()
  let count = 0
  
  snapshot.forEach((doc) => {
    const data = doc.data()
    if (data.location) {
      const geohash = geohashForLocation([
        data.location.latitude,
        data.location.longitude
      ])
      
      batch.update(doc.ref, { geohash })
      count++
      
      // Firestore batch limit is 500
      if (count === 500) {
        batch.commit()
        count = 0
      }
    }
  })
  
  if (count > 0) {
    await batch.commit()
  }
  
  console.log('Migration complete')
}
```

## Performance Optimization

<Steps>
  <Step title="Use Composite Indexes">
    Create composite indexes for frequently used query combinations in Firebase Console
  </Step>
  
  <Step title="Implement Pagination">
    Always use pagination for large result sets to reduce read costs and improve performance
  </Step>
  
  <Step title="Cache Data Locally">
    Enable offline persistence to reduce database reads
    
    ```javascript
    import { enableIndexedDbPersistence } from 'firebase/firestore'
    import { db } from './firebase'

    enableIndexedDbPersistence(db).catch((err) => {
      if (err.code === 'failed-precondition') {
        console.log('Persistence failed: multiple tabs open')
      } else if (err.code === 'unimplemented') {
        console.log('Persistence not supported')
      }
    })
    ```
  </Step>
  
  <Step title="Denormalize Data">
    Duplicate frequently accessed data to avoid multiple reads
  </Step>
</Steps>
